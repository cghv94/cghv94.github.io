Modelo stepwise both sides

empty.model <- lm(Linea_1 ~ 1, data=ingresos)
horizonte <- formula(Linea_1 ~ Linea_2 + Linea_3 + Linea_4 + Linea_5 + Linea_6 + Linea_7 + Linea_8 + Linea_9 + Linea_12 + Linea_A + Linea_B)
stepwise <- stepAIC(empty.model, trace=FALSE, direction="both", scope=horizonte)
summary(stepwise)
